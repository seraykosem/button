## Install

```bash
npm install --save my-button-component
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'my-button-component'
import 'my-button-component/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
